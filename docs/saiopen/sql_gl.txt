SELECT
    G.FECHA,
    C.ID_N,
    C.COMPANY,
    A.CDGOTTL AS TITULO,
    ACT.DESCRIPCION AS NOMBRE_TITULO,
    A.CDGOGRPO AS GRUPO,
    ACG.DESCRIPCION AS NOMBRE_GRUPO,
    A.CDGOCNTA AS CUENTA,
    ACC.DESCRIPCION AS NOMBRE_CUENTA,
    A.CDGOSBCNTA AS SUBCUENTA,
    ACS.DESCRIPCION AS NOMBRE_SUBCUENTA,
    G.ACCT AS AUXILIAR,
    A.DESCRIPCION AS NOMBRE_AUXILIAR,
    G.DEBIT,
    G.CREDIT,
    G.TIPO,
    G.BATCH,
    G.DEPTO AS DEPARTAMENTO,
    LD.DESCRIPCION AS NOMBRE_DEPARTAMENTO,
    G.CCOST AS CENTRO_COSTO,
    LC.DESCRIPCION AS NOMBRE_CENTRO_COSTO,
    G.PROYECTO,
    P.DESCRIPCION AS NOMBRE_PROYECTO,
    G.ACTIVIDAD,
    AC.DESCRIPCION AS NOMBRE_ACTIVIDAD
FROM GL G
INNER JOIN CUST C ON C.ID_N = G.ID_N
INNER JOIN ACCT A ON A.ACCT = G.ACCT
INNER JOIN ACCT ACT On A.CDGOTTL = ACT.ACCT
INNER JOIN ACCT ACG On A.CDGOGRPO = ACG.ACCT
INNER JOIN ACCT ACC On A.CDGOCNTA = ACC.ACCT
INNER JOIN ACCT ACS On A.CDGOSBCNTA = ACS.ACCT
LEFT JOIN LISTA LD On LD.CODIGO = G.DEPTO AND LD.TIPO='DP'
LEFT JOIN LISTA LC On LC.CODIGO = G.CCOST AND LC.TIPO='CC'
LEFT JOIN PROYECTOS P On P.CODIGO = G.PROYECTO
LEFT JOIN ACTIVIDADES AC On Ac.CODIGO = G.ACTIVIDAD
